import random

API_KEYS = [
    os.environ.get("GEMINI_API_KEY_1"),
    os.environ.get("GEMINI_API_KEY_2"),
    os.environ.get("GEMINI_API_KEY_3"),
    os.environ.get("GEMINI_API_KEY_4"),
    os.environ.get("GEMINI_API_KEY"), 
]
# Filter out None values just in case
API_KEYS = [k for k in API_KEYS if k]

if not API_KEYS:
    print("WARNING: No GEMINI_API_KEYS found in environment variables.")

MODEL_NAME = 'gemini-2.5-flash' #main model 
#MODEL_NAME = 'gemini-3-flash-preview' #DIAGRAM
#MODEL_NAME = 'gemini-3.1-pro-preview' #LIMIT

def get_configured_model():
    """
    Selects a random API key, configures GenAI, increments usage stats, 
    and returns the model instance.
    """
    if not API_KEYS:
        raise Exception("No API Keys available.")
    
    selected_key_index = random.randint(0, len(API_KEYS) - 1)
    selected_key = API_KEYS[selected_key_index]
    key_name = f"key_{selected_key_index + 1}"
    
    # Configure GenAI with the selected key
    genai.configure(api_key=selected_key)
    
    # Track usage in Firestore
    try:
        db = firestore.client()
        stats_ref = db.collection('stats').document('api_usage')
        stats_ref.set({key_name: firestore.Increment(1)}, merge=True)
        print(f"Using {key_name} ... Usage tracked.")
    except Exception as e:
        print(f"Error tracking API usage: {e}")
    return genai.GenerativeModel(model_name=MODEL_NAME)

def upload_to_cloudinary(file_obj, filename):
    try:
        clean_filename = "".join([c if c.isalnum() or c in "._-" else "_" for c in filename])
        
        # Determine resource type: 'raw' for PDFs to avoid image transformation strictness, 'auto' otherwise
        res_type = "raw" if filename.lower().endswith(".pdf") else "auto"

        print(f"Uploading {clean_filename} to Cloudinary as {res_type}...")
        response = cloudinary.uploader.upload(
            file_obj, 
            resource_type = res_type,
            folder = "ktu_notes",
            public_id = clean_filename.split('.')[0] 
        )
        
        web_url = response.get("secure_url")
        print(f"File uploaded to Cloudinary: {web_url}")
        return web_url, response.get("public_id")

    except Exception as e:
        print(f"Cloudinary Upload Error: {e}")
        raise e

def clean_mermaid_code(code):
    code = re.sub(r'```mermaid|```', '', code).strip()
    return code

def extract_text_from_pdf_stream(file_stream):
    try:
        with fitz.open(stream=file_stream.read(), filetype="pdf") as doc:
            text = ""
            for page in doc:
                text += page.get_text()
        file_stream.seek(0) # Reset stream
        return text
    except Exception as e:
        print(f"PDF Extract Error: {e}")
        return ""

@app.route('/verify-note', methods=['POST'])
def verify_note():
    try:
        if 'file' not in request.files:
            return jsonify({"error": "No file part"}), 400
        file = request.files['file']
        subject = request.form.get('subject', 'Unknown Subject')
        module = request.form.get('module', 'Unknown Module')
        if file.filename == '':
            return jsonify({"error": "No selected file"}), 400
        # Save to temp file to avoid RAM OOM on Render
        filename = file.filename
        temp_path = UPLOAD_FOLDER / filename
        file.save(str(temp_path))
        print(f"Saved temp file to: {temp_path}")
        # Check Global Settings for AI Verification
        ai_enabled = True # Default
        try:
            db = firestore.client()
            settings_ref = db.collection('config').document('settings')
            doc = settings_ref.get()
            if doc.exists:
                ai_enabled = doc.to_dict().get('enableAiVerification', True)
        except Exception as db_e:
            print(f"Error fetching settings: {db_e}. Defaulting to AI ON.")
        extracted_text = ""
        status = "pending"
        reason = "AI Verification Failed or Skipped"
        ai_summary = "No summary generated."
        # If AI is disabled, skip extraction and verification
        if not ai_enabled:
             print("AI Verification is DISABLED by Admin. Skipping...")
             status = "pending"
             reason = "Manual Mode Active (AI Verification Disabled)"
             ai_summary = "Pending Admin Review"
        else:
            # AI Verification Logic
            try:
                # Extract text using file path (memory efficient)
                with fitz.open(temp_path) as doc:
                    for page in doc:
                        extracted_text += page.get_text()
                if extracted_text.strip():
                    model = get_configured_model()
                    prompt = f"""
                    Act as a Syllabus Validator for KTU Engineering Course.
                    Subject: {subject}
                    Module: {module}
                    Content of the Note:
                    {extracted_text[:10000]}
                    Task:
                    1. Verify if the content is relevant to the Subject and Module provided.
                    2. If relevant, status is "approved". If completely irrelevant (spam, wrong subject), status is "rejected". If unsure or partially correct, status is "pending".
                    3. Generate a short 2-sentence summary of the note.
                    CRITICAL: Do not include ANY greetings, introductions, or conversational filler in the summary.
                    Return ONLY JSON:
                    {{
                        "status": "approved" | "rejected" | "pending",
                        "reason": "Explanation for the decision",
                        "summary": "Short summary of the content"
                    }}
                    """
                    try:
                        response = model.generate_content(prompt)
                        clean_json = re.sub(r'```json|```', '', response.text).strip()
                        ai_data = json.loads(clean_json)
                        status = ai_data.get('status', 'pending')
                        reason = ai_data.get('reason', 'AI Review')
                        ai_summary = ai_data.get('summary', 'Summary not found')
                    except Exception as ai_e:
                        print(f"AI Error: {ai_e}")
                        status = "pending"
                        reason = "AI Processing Error, marked as pending for human review."
            except Exception as e:
                print(f"Extraction Error: {e}")
        # Upload to Cloudinary
        print("Uploading to Cloudinary...")
        # Upload from path
        file_url, file_path_id = upload_to_cloudinary(str(temp_path), filename) # Cloudinary python SDK supports path string
        # Clean up temp file
        try:
            if temp_path.exists():
                os.remove(temp_path)
                print(f"Removed temp file: {temp_path}")
        except Exception as cleanup_e:
            print(f"Cleanup Error: {cleanup_e}")
        return jsonify({
            "status": status,
            "reason": reason,
            "summary": ai_summary,
            "url": file_url,
            "fileId": file_path_id
        })
    except Exception as e:
        print(f"Verify Note Error: {e}")
        # Attempt minimal cleanup
        try:
             if 'temp_path' in locals() and temp_path.exists():
                 os.remove(temp_path)
        except: pass
        return jsonify({"error": str(e)}), 500

def extract_text_from_url(url):
    headers = {
        'User-Agent': 'Mozilla/5.0'
    }
    try:
        response = requests.get(url, headers=headers, timeout=20)
        if response.status_code == 200:
            with fitz.open(stream=response.content, filetype="pdf") as doc:
                text = ""
                for page in doc:
                    text += page.get_text()
            return text
        else:
            raise Exception(f"Download failed with status: {response.status_code}")
    except Exception as e:
        raise Exception(f"Failed to extract text from PDF: {str(e)}")

@app.route('/smart-search', methods=['POST'])
def smart_search():
    try:
        data = request.get_json()
        query = data.get('query', '').strip()
        branch = data.get('branch', '')
        semester = data.get('semester', '')
        subject = data.get('subject', '')
        if not query:
            return jsonify({"error": "No search query provided"}), 400
        print(f"Executing Smart Search for: '{query}' in {subject} ({branch})...")
        # 1. Search Firestore for related notes
        found_notes = []
        try:
             db = firestore.client()
             notes_ref = db.collection('notes')
             # Firestore doesn't have native full-text search easily without external extensions,
             # so we do a basic prefix/exact match or fetch recent notes and filter in python.
             # For a robust small app, fetching all approved notes for the sem/branch and filtering in memory is okay.
             base_query = notes_ref.where('status', '==', 'approved')
             if semester and semester != "Not Set":
                 base_query = base_query.where('semester', '==', semester)
             if branch and branch != "Not Set":
                  base_query = base_query.where('branch', '==', branch)
             docs = base_query.stream()
             for doc in docs:
                 note_data = doc.to_dict()
                 # Check against subject, module, or summary (basic fallback text-search)
                 subject = str(note_data.get('subject', ''))
                 module = str(note_data.get('module', ''))
                 summary = str(note_data.get('summary', ''))
                 found_notes.append({
                     "id": doc.id,
                     "subject": subject,
                     "module": module,
                     "summary": summary,
                     "url": note_data.get('url'),
                     "uploaderId": note_data.get('uploaderId'),
                     "timestamp": note_data.get('timestamp')
                 })
             # Sort notes by newest (basic sort if timestamp exists)
             if found_notes and 'timestamp' in found_notes[0]:
                  found_notes = sorted(found_notes, key=lambda x: str(x.get('timestamp', '')), reverse=True)
        except Exception as db_e:
             print(f"Firestore Search Error: {db_e}")
             # Non-fatal, we can still return AI results
        # Build Summary Context for Gemini
        summaries_context = ""
        for i, note in enumerate(found_notes[:15]): # Cap to latest 15 to avoid massive prompts
            summaries_context += f"Note ID: {note['id']}\nSubject: {note['subject']} {note['module']}\nSummary: {note['summary']}\n\n"
        # 2. Call Gemini for Definitions, Related Qs, PYQs, and Best Note Matching
        model = get_configured_model()
        prompt = f"""
        Act as a KTU (Kerala Technological University) Professor for the subject: {subject if subject else 'General'}.
        A student is searching for the topic: "{query}" in the context of {branch} {semester}.
        Below is a list of available PDF Notes and their AI-generated summaries:
        {summaries_context if summaries_context else "No notes currently available in the database."}
        Task:
        1. Identify which Note from the list above is the **best match** for the topic "{query}".
        2. Set "bestNoteId" to exactly that Note's "Note ID" string. If no match, return null.
        3. In the "definitions" field, provide a clear summary of the topic. **Do NOT mention the PDF name here**.
        4. Generate 3 related questions and 2-3 PYQs.
        Return ONLY raw JSON.
        Format:
        {{
            "bestNoteId": "NOTE_ID_FROM_LIST",
            "definitions": "Summary...",
            "relatedQuestions": [...],
            "pyqs": [...]
        }}
        """
        response = model.generate_content(prompt)
        clean_json = re.sub(r'```json|```', '', response.text).strip()
        ai_data = json.loads(clean_json)
        print(f"DEBUG: Smart Search AI Data: {json.dumps(ai_data, indent=2)}")
        best_note_id = ai_data.get('bestNoteId')
        source_note_name = None
        matched_notes_list = []
        # Robust Matching Logic
        if best_note_id and best_note_id != "null":
            target_id = str(best_note_id).strip().lower()
            # Step 1: Direct ID Match
            for n in found_notes:
                curr_id = str(n['id']).strip().lower()
                if target_id == curr_id or target_id in curr_id or curr_id in target_id:
                    matched_notes_list.append(n)
                    break
            # Step 2: Fallback to Subject/Module keyword match if ID match failed
            if not matched_notes_list:
                for n in found_notes:
                    subj_mod = f"{n['subject']} {n['module']}".lower()
                    if target_id in subj_mod or subj_mod in target_id:
                        matched_notes_list.append(n)
                        break
            if matched_notes_list:
                best_match = matched_notes_list[0]
                best_note_id = best_match['id'] # Set it to the real ID for the frontend
                subj = best_match.get('subject')
                mod = best_match.get('module')
                source_note_name = f"{subj} - {mod}" if subj and mod else subj
                print(f"DEBUG: Successfully matched note: {source_note_name}")
        # Final Fallback: Basic text search if still no match
        if not matched_notes_list and found_notes:
            query_lower = query.lower()
            for n in found_notes:
                if query_lower in str(n['subject']).lower() or query_lower in str(n['module']).lower():
                    matched_notes_list.append(n)
                    best_match = n
                    best_note_id = best_match['id']
                    subj = best_match.get('subject')
                    mod = best_match.get('module')
                    source_note_name = f"{subj} - {mod}" if subj and mod else subj
                    break
        result = {
            "notes": matched_notes_list,
            "sourceNoteName": source_note_name,
            "bestNoteId": best_note_id,
            "definitions": ai_data.get('definitions', 'No definitions available.'),
            "relatedQuestions": ai_data.get('relatedQuestions', []),
            "pyqs": ai_data.get('pyqs', [])
        }
        # Write debug info to a file since terminal output is hard to read
        try:
            with open('C:/temp/smart_search_debug.json', 'w') as f:
                json.dump({
                    "query": query,
                    "best_note_id": best_note_id,
                    "source_note_name": source_note_name,
                    "matched_count": len(matched_notes_list)
                }, f, indent=2)
        except Exception as e:
            print(f"Error writing debug file: {e}")
        return jsonify(result)
    except Exception as e:
        print(f"Smart Search Error: {str(e)}")
        return jsonify({"error": str(e)}), 500

        query = data.get('query', '').strip()
        history = data.get('history', [])
        branch = data.get('branch', '')
        semester = data.get('semester', '')
        if not query:
            return jsonify({"error": "No query provided"}), 400
        print(f"Executing AI Doubt Resolver for: '{query}' with history length: {len(history)}")
        # 1. Fetch relevant notes from Firestore
        db = firestore.client()
        notes_ref = db.collection('notes')
        # Filter by branch and semester
        base_query = notes_ref.where('status', '==', 'approved')
        if semester and semester != "Not Set":
            base_query = base_query.where('semester', '==', semester)
        if branch and branch != "Not Set":
            base_query = base_query.where('branch', '==', branch)
        docs = base_query.stream()
        context_parts = []
        for doc in docs:
            note_data = doc.to_dict()
            subject = note_data.get('subject', 'Unknown Subject')
            module = note_data.get('module', 'Unknown Module')
            summary = note_data.get('summary', '')
            # Use summary as context. In a full RAG, we'd use vector search on chunks.
            # But for this MVP, summaries and metadata are high-value context.
            context_parts.append(f"Subject: {subject}, Module: {module}\nContent Summary: {summary}")
        context_str = "\n\n".join(context_parts[:20]) # Cap to avoid token limits
        # 2. Call Gemini with RAG Prompt
        model = get_configured_model()
        # Format history for Gemini
        formatted_history = []
        for msg in history:
            # Skip messages without text or unexpected roles
            if not msg.get('text') or msg.get('role') not in ['user', 'ai']:
                continue
            formatted_history.append({
                "role": "user" if msg['role'] == 'user' else "model",
                "parts": [msg['text']]
            })
        # CRITICAL: Gemini requires the first message in history to be from the 'user'
        while formatted_history and formatted_history[0]['role'] != 'user':
            formatted_history.pop(0)
        print(f"DEBUG: Formatted history (after fixing start): {json.dumps(formatted_history, indent=2)}")
        system_instruction = f"""
        Act as a helpful KTU (Kerala Technological University) Tutor. 
        **CRITICAL TASK**: 
        Identify which Subject and Module from the provided context below contains the information about the student's doubt.
        Always start your response by clearly stating the Subject and Module if found.
        Example: "This topic is covered in **[Subject]**, **[Module]**. According to the notes, ..."
        Use the following information extracted from the course notes as your primary source.
        Context from Uploaded Notes:
        {context_str if context_str else "No specific notes found for this branch/semester yet."}
        Task:
        - Provide a clear, concise, and accurate answer to the student's question.
        - Start directly with the answer and the source reference.
        - If the info is found in the notes, say: "Refer to the PDF for **[Subject] - [Module]**."
        - Avoid ANY introductory or concluding remarks (no "Sure!", "Hello", "I hope this helps").
        - Use Markdown for formatting (bolding, lists, etc.) to improve readability.
        - Keep the total response under 150-200 words.
        Format: Clean natural language message, starting DIRECTLY with the answer and source reference.
        """
        # Use Chat Session for memory
        chat = model.start_chat(history=formatted_history)
        # We include the context and system instruction in each message to ensure it stays relevant,
        # but the chat session handles the conversational flow.
        full_query = f"{system_instruction}\n\nStudent's Query: {query}"
        response = chat.send_message(full_query)
        return jsonify({
            "answer": response.text.strip(),
        })
    except Exception as e:
        print(f"Ask Doubt Error: {str(e)}")
        return jsonify({"error": str(e)}), 500

        return jsonify({"error": str(e)}), 500

import time
import traceback
from functools import wraps

# Performance Monitor Decorator
def monitor_performance(threshold_seconds=10):
    def decorator(f):
        @wraps(f)
        def wrapper(*args, **kwargs):
            start_time = time.time()
            try:
                result = f(*args, **kwargs)
                return result
            except Exception as e:
                # Let the global error handler catch exceptions
                raise e
            finally:
                execution_time = time.time() - start_time
                if execution_time > threshold_seconds:
                    endpoint = request.path
                    print(f"⚠️ Slow Performance Alert: Endpoint {endpoint} took {execution_time:.2f}s (Threshold: {threshold_seconds}s)")
        return wrapper
    return decorator

# Global Error Handler
@app.errorhandler(Exception)
def handle_global_exception(e):
    # Get traceback
    tb_str = traceback.format_exc()
    endpoint = request.path
    method = request.method
    
    alert_msg = (
        f"🚨 **CRASH ALERT** 🚨\n\n"
        f"📍 **Endpoint:** `{method} {endpoint}`\n"
        f"❌ **Error:** `{str(e)}`\n\n"
        f"📜 **Traceback:**\n```python\n{tb_str[:1500]}...```" # Truncate if too long
    )
    
    print(alert_msg)
    
    # Return standard error to client
    print(f"Global Error Caught: {e}")
    return jsonify({"error": "Internal Server Error", "details": str(e)}), 500
